def visualize(filename):
    "Plot the Features"
    data = pd.read_csv(filepath_or_buffer = filename)
    average = data.mean(axis=0)
    minimum = data.min(axis=0)
    maximum = data.max(axis=0)
    fig = plt.figure(figsize = (12,12))
    x = list(range(1,41))
    plt.plot(x, average, color = 'blue', label = 'average')
    plt.plot(x, minimum, color = 'green', label ='minimum')
    plt.plot(x, maximum, color ='red', label ='maximum')
    plt.xlabel('Feature#')
    plt.ylabel('Values')
    plt.show()
